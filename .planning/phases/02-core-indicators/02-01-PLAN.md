3. **calculate_adx(klines, period=14)**: Use pandas-ta adx() - returns ADX, DI+, DI-
4. **calculate_pivot_channel(highs, lows, period=20)**: Rolling max/min shifted by 1
5. **calculate_donchian_channel(highs, lows, period=20)**: Classic Donchian
6. **calculate_linear_regression_channel(close, period=20)**: LR upper/lower bands

Key points:
- Normalize klines to DataFrame with lowercase columns: time, open, high, low, close, volume
- Use ta.mfi(df['high'], df['low'], df['close'], df['volume'], length=period)
- Use ta.cci(df['high'], df['low'], df['close'], length=period)
- Use ta.adx(df['high'], df['close'], df['low'], length=period) - note column order!
- Return result as list of dicts with indicator values
- Include all OHLCV fields alongside indicators

```python
def calculate_indicators(klines: List, mfi_period=14, cci_period=20, adx_period=14, channel_period=20, channel_type='pivot') -> List[dict]:
    df = pd.DataFrame(klines, columns=['time', 'open', 'high', 'low', 'close', 'volume'])
    for col in ['open', 'high', 'low', 'close', 'volume']:
        df[col] = pd.to_numeric(df[col])
    
    # MFI
    df['mfi'] = ta.mfi(df['high'], df['low'], df['close'], df['volume'], length=mfi_period)
    
    # CCI
    df['cci'] = ta.cci(df['high'], df['low'], df['close'], length=cci_period)
    
    # ADX, DI+, DI-
    adx_result = ta.adx(df['high'], df['close'], df['low'], length=adx_period)
    df['adx'] = adx_result[f'ADX_{adx_period}']
    df['di_plus'] = adx_result[f'DMP_{adx_period}']
    df['di_minus'] = adx_result[f'DMN_{adx_period}']
    
    # Price Channel
    if channel_type == 'pivot':
        df['channel_upper'] = df['high'].rolling(channel_period).max().shift(1)
        df['channel_lower'] = df['low'].rolling(channel_period).min().shift(1)
    elif channel_type == 'donchian':
        df['channel_upper'] = df['close'].rolling(channel_period).max()
        df['channel_lower'] = df['close'].rolling(channel_period).min()
    elif channel_type == 'linear_regression':
        # Simple linear regression channel
        df['channel_middle'] = df['close'].rolling(channel_period).mean()
        std = df['close'].rolling(channel_period).std()
        df['channel_upper'] = df['channel_middle'] + (std * 2)
        df['channel_lower'] = df['channel_middle'] - (std * 2)
    
    return df.to_dict(orient='records')
```

```python
from fastapi import APIRouter, Query
from typing import Optional
from app.services.indicators import calculate_indicators
from app.models.indicators import IndicatorResponse

router = APIRouter()

@router.get("/indicators", response_model=IndicatorResponse)
async def get_indicators(
    symbol: str = Query("BTC/USDT", description="Trading symbol"),
    interval: str = Query("1h", description="Timeframe"),
    mfi_period: int = Query(14, description="MFI period"),
    cci_period: int = Query(20, description="CCI period"),
    adx_period: int = Query(14, description="ADX period"),
    channel_period: int = Query(20, description="Channel period"),
    channel_type: str = Query("pivot", description="Channel type: pivot, donchian, linear_regression"),
    limit: int = Query(200, description="Number of candles"),
) -> IndicatorResponse:
    """Calculate MFI, CCI, ADX, DI, and Price Channel indicators."""
    from app.services.binance import fetch_klines
    
    klines = await fetch_klines(symbol, interval, limit=limit + max(mfi_period, cci_period, adx_period, channel_period))
    data = calculate_indicators(
        klines,
        mfi_period=mfi_period,
        cci_period=cci_period,
        adx_period=adx_period,
        channel_period=channel_period,
        channel_type=channel_type,
    )
    
    return IndicatorResponse(
        symbol=symbol,
        interval=interval,
        data=data[-limit:],  # Return only requested number
        count=len(data[-limit:])
    )
```
